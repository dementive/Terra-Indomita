me_thaton.131 = { # Shenli Response
    type = country_event
    title = me_thaton_131_title
    desc = me_thaton_131_desc
    picture = interesting_histories_greek_debate

    immediate = {
        if = {
            limit = {
                has_variable = thaton_sent_us_da_cash
            }
            add_treasury = 50
            current_ruler = {
                add_gold = 100
            }
            set_variable = {
                name = thaton_ultimatum_decision_var
                value = {
                    integer_range = {
                        min = 3
                        max = 6
                    }
                }
                days = 180
            }
        }
        else = {
            limit = {
                NOT = { has_variable = thaton_sent_us_da_cash }
            }
            set_variable = {
                name = thaton_ultimatum_decision_var
                value = {
                    integer_range = {
                        min = 1
                        max = 5
                    }
                }
                days = 180
            }
        }
    }

    option = {
        name = me_thaton_131.a
        #We will accept their alliance offer
        make_subject = {
            target = c:SHE
            type = satrapy
        }
        c:THT = {
            set_variable = {
                name = shenli_accepted_ultimatum
                days = 180
            }
            trigger_event = {
                id = me_thaton.132
                days = { 15 30 }
            }
        }
        ai_chance = {
            base = 5
            modifier = {
                factor = -50
                var:thaton_ultimatum_decision_var <= 3
            }
            modifier = {
                factor = 50
                var:thaton_ultimatum_decision_var >= 4
            }
        }
    }
    option = {
        name = me_thaton_131.b
        #Offer them to become a tributary (ai only)
        trigger = {
            c:SHE = { is_ai = yes }
        }
        c:SHE = {
            declare_war_with_wargoal = {
                target = c:THT
                war_goal = imperial_conquest_wargoal
            }
        }
        ai_chance = {
            base = 5
            modifier = {
                factor = 25
                var:thaton_ultimatum_decision_var <= 3
            }
            modifier = {
                factor = -25
                var:thaton_ultimatum_decision_var >= 4
            }
        }
    }
    option = {
        name = me_thaton_131.c
        #Egypt wants nothing to do with Rhodes!
        add_political_influence = 5
        c:THT = {
            set_variable = {
                name = shenli_does_nothing
                days = 180
            }
            trigger_event = {
                id = me_thaton.132
                days = { 15 30 }
            }
        }
        ai_chance = {
            base = 5
            modifier = {
                factor = 50
                var:thaton_ultimatum_decision_var <= 3
            }
            modifier = {
                factor = -50
                var:thaton_ultimatum_decision_var >= 4
            }
        }
    }
}
